## Test permission override
python - <<'PY'
import requests

BASE='http://127.0.0.1:8000'
ADMIN_EMAIL='learnez@email.com'
ADMIN_PASSWORD='123456'
TARGET_EMAIL='lecturer1@email.com'

def req(method,path,token=None,body=None,params=None):
    h={}
    if token:
        h['Authorization']=f'Bearer {token}'
    if body is not None:
        h['Content-Type']='application/json'
    r=requests.request(method,BASE+path,headers=h,json=body,params=params,timeout=20)
    try:
        data=r.json()
    except Exception:
        data=r.text
    print(f'{method} {path} -> {r.status_code}')
    print(data)
    print('-'*70)
    return r.status_code,data

# login
code,data=req('POST','/api/iam/login',body={'email':ADMIN_EMAIL,'password':ADMIN_PASSWORD})
if code!=200:
    raise SystemExit('login failed')
token=data['access_token']

# make sure catalog and role defaults exist
req('POST','/api/iam/rbac/permissions/sync',token=token)

# get target user_id
code,acc=req('GET','/api/iam/accounts/',token=token)
row=next((u for u in acc if u.get('email')==TARGET_EMAIL),None)
if not row:
    raise SystemExit('target user not found')
uid=row['id']
print('TARGET USER ID:',uid)
print('-'*70)

# base permissions from role
req('GET',f'/api/iam/rbac/users/{uid}/permissions',token=token)

# apply overrides: revoke assignment-04 (13), explicitly allow module-02 (3)
overrides_payload={
  'overrides':[
    {'permission_id':13,'is_allowed':False},
    {'permission_id':3,'is_allowed':True}
  ]
}
req('PUT',f'/api/iam/rbac/users/{uid}/permission-overrides',token=token,body=overrides_payload)

# verify override rows
req('GET',f'/api/iam/rbac/users/{uid}/permission-overrides',token=token)

# verify effective permissions after override (13 should disappear)
req('GET',f'/api/iam/rbac/users/{uid}/permissions',token=token)

# verify role defaults unchanged (lecturer role still has 13)
req('GET','/api/iam/rbac/roles/2/permissions',token=token)
PY
